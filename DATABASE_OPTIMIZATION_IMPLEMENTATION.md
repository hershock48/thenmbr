const { data, error, metrics } = await db.select('subscribers', {
  useCache: true,
  cacheTTL: 5 * 60 * 1000,
  cacheTags: ['subscribers', `org:${orgId}`],
  selectFields: ['id', 'email', 'first_name', 'last_name']

  org_id: orgId,
  status: 'active'
})

  useCache: false,

})
